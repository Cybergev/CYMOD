demobilization = {
	demob_manpower_mission = {

		icon = revolt

		is_good = no

		days_mission_timeout = 30

		cancel_trigger = {
			OR = {
				has_unsupported_manpower_law = no
				has_war = yes
			}
		}

		activation = {
			has_unsupported_manpower_law = yes
			has_war = no
			if = {
				limit = {
					original_tag = GRE
				}
				NOT = {
					has_completed_focus = GRE_request_communist_support
				}
			}
		}

		cancel_if_not_visible = yes
		selectable_mission = yes
		complete_effect = {
			if = {
				limit = {
					NOT = {
						has_idea = volunteer_only
					}
					has_war_support < 0.1
				}
				add_ideas = volunteer_only
			}
			else_if = {
				limit = {
					NOT = {
						has_idea = limited_conscription
					}
					has_war_support < 0.2
				}
				add_ideas = limited_conscription
			}
			else_if = {
				limit = {
					NOT = {
						has_idea = extensive_conscription
					}
					has_war_support < 0.6
				}
				add_ideas = extensive_conscription
			}
			else = {
				add_ideas = service_by_requirement
			}
			if = {
				limit = { 
					has_idea = mutiny_1
				}
				remove_ideas = mutiny_1
			}
			else_if = {
				limit = {
					has_idea = mutiny_2
				}
				remove_ideas = mutiny_2
			}
			else = {
				remove_ideas = mutiny_3
			}
		}
		timeout_effect = {
			if = {
				limit = {
					has_government = communism
					has_stability < 0.2
				}
				start_civil_war = {
					ideology = democratic
				}
			}
			if = {
				limit = {
					NOT = { has_government = communism }
					has_stability < 0.3
				}
				start_civil_war = {
					ideology = communism
				}
			}
			if = {
				limit = {
					has_stability > 0.2
				}
				add_stability = -0.1
			}
			if = {
				limit = {
					has_government = communism
				}
				add_popularity = { ideology = democratic popularity = 0.1 }
			}
			if = {
				limit = {
					NOT = { has_government = communism }
				}
				add_popularity = { ideology = communism popularity = 0.1 }
			}
		}

		ai_will_do = {
			factor = 500
		}
	}
	demob_economic_mission = {

		icon = revolt
		
		is_good = no

		selectable_mission = yes

		cancel_trigger = {
			AND = {
				has_war = yes
				OR = {
					has_unsupported_economic_law = no
					AND = {
						has_idea = Flexible_economy
						NOT = {
							has_war_support < Flexible_economy_policy_military_share
						}
					}
				}
			}
		}

		activation = {
			AND = {
				has_war = no
				OR = {
					has_unsupported_economic_law = yes
					AND = {
						has_idea = Flexible_economy
						has_war_support < Flexible_economy_policy_military_share
					}
				}
			}
		}

		days_mission_timeout = 30

		complete_effect = {
			if = {
				limit = {
					NOT = {
						has_dynamic_modifier = {
							modifier = Flexible_economy_policy
						}
					}
				}
				else_if = {
					limit = {
						NOT = {
							has_idea = low_economic_mobilisation
						}
						has_war_support > 0.15
						has_war_support < 0.25
					}
					add_ideas = low_economic_mobilisation
				}
				else_if = {
					limit = {
						NOT = {
							has_idea = partial_economic_mobilisation
						}
						has_war_support > 0.25
						has_war_support < 0.50
					}
					add_ideas = partial_economic_mobilisation
				}
				else_if = {
					limit = {
						NOT = {
							has_idea = war_economy
						}
						has_war_support > 0.50
						has_war_support < 0.80
					}
					add_ideas = war_economy
				}
			}
			if = {
				limit = {
					has_dynamic_modifier = {
						modifier = Flexible_economy_policy
					}
				}
				set_temp_variable = {
					demob_value = has_war_support
				}
				subtract_from_temp_variable = {
					demob_value = Flexible_economy_policy_military_share
				}
				set_variable = {
					increasing_military_economy_value = demob_value
				}
				increase_military_economy = yes
			}
			if = {
				limit = {
					has_idea = strikes_1
				}
				remove_ideas = strikes_1
			}
			if = {
				limit = {
					has_idea = strikes_2
				}
				remove_ideas = strikes_2
			}
			if = {
				limit = {
					has_idea = strikes_3
				}
				remove_ideas = strikes_3
			}
		}
		timeout_effect = {
			activate_mission = demob_economic_mission
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.21
					NOT = {
						has_government = communism
					}
				}
				start_civil_war = {
					ideology = communism
				}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.21
					has_government = communism
					
				}
				start_civil_war = {
					ideology = democratic
				}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.41
					NOT = {
						has_government = communism
					}
				}
				add_popularity = { ideology = communism popularity = 0.1}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.41
					has_government = communism
					
				}
				add_popularity = { ideology = democratic popularity = 0.1}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability > 0.4
				}
				add_stability = -0.1
			}
			if = {
